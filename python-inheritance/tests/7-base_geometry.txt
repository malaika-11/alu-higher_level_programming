# 7-base_geometry tests

# Test 1: Check instantiation
python3 -c "from 7-base_geometry import BaseGeometry; bg = BaseGeometry()"

# Test 2: Check area() raises Exception
python3 -c "from 7-base_geometry import BaseGeometry; bg = BaseGeometry(); bg.area()" 2>&1 | grep "area() is not implemented"

# Test 3: Check integer_validator() with valid integers
python3 -c "from 7-base_geometry import BaseGeometry; bg = BaseGeometry(); bg.integer_validator('my_int', 12); bg.integer_validator('width', 89)"

# Test 4: Check integer_validator() raises TypeError for non-integer
python3 -c "from 7-base_geometry import BaseGeometry; bg = BaseGeometry(); bg.integer_validator('name', 'John')" 2>&1 | grep "TypeError: name must be an integer"

# Test 5: Check integer_validator() raises ValueError for 0
python3 -c "from 7-base_geometry import BaseGeometry; bg = BaseGeometry(); bg.integer_validator('age', 0)" 2>&1 | grep "ValueError: age must be greater than 0"

# Test 6: Check integer_validator() raises ValueError for negative
python3 -c "from 7-base_geometry import BaseGeometry; bg = BaseGeometry(); bg.integer_validator('distance', -4)" 2>&1 | grep "ValueError: distance must be greater than 0"
